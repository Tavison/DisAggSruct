119template<typename T, std::size_t... Is>
120constexpr bool try_construct_n(std::index_sequence<Is...>) {
121 return requires { T{ (static_cast<void>(Is), any_type{})... }; };

132constexpr bool constructible_with_n = try_construct_n<T>(std::make_index_sequence<N>{});

149template<typename T, std::size_t N = 0>
150constexpr std::size_t count_fields() {
151 if constexpr (constructible_with_n<T, N + 1>)
152 return count_fields<T, N + 1>();

201template<typename T>
202auto to_tuple(T& s) {
203 constexpr std::size_t N = field_count_v<T>;
204 static_assert(N <= 16, "to_tuple supports structs with up to 16 fields");
205
206 if constexpr (N == 0) { return std::tuple<>{}; }
207 else if constexpr (N == 1) { auto& [a] = s; return std::tie(a); }
208 else if constexpr (N == 2) { auto& [a,b] = s; return std::tie(a,b); }
209 else if constexpr (N == 3) { auto& [a,b,c] = s; return std::tie(a,b,c); }
210 else if constexpr (N == 4) { auto& [a,b,c,d] = s; return std::tie(a,b,c,d); }
211 else if constexpr (N == 5) { auto& [a,b,c,d,e] = s; return std::tie(a,b,c,d,e); }
212 else if constexpr (N == 6) { auto& [a,b,c,d,e,f] = s; return std::tie(a,b,c,d,e,f); }
213 else if constexpr (N == 7) { auto& [a,b,c,d,e,f,g] = s; return std::tie(a,b,c,d,e,f,g); }
214 else if constexpr (N == 8) { auto& [a,b,c,d,e,f,g,h] = s; return std::tie(a,b,c,d,e,f,g,h); }

249T from_tuple(Tuple&& t) {

285template<typename Tuple, typename Fn>
286void for_each_field(Tuple& t, Fn&& fn) {
287 constexpr std::size_t N = std::tuple_size_v<std::remove_reference_t<Tuple>>;
288 [&]<std::size_t... Is>(std::index_sequence<Is...>) {
289 (fn(std::get<Is>(t)), ...);
290 }(std::make_index_sequence<N>{});
291}

329template<typename T, typename Policy>
330T disaggregate(T s, Policy policy) {
331 auto t = to_tuple(s);
332 for_each_field(t, [&policy](auto& field) { policy(field); });
333 return from_tuple<T>(t);
334}
